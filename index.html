// api/picks.js // Fetches today's ESPN schedule and auto-generates model picks export default async function handler(req, res) { res.setHeader('Access-Control-Allow-Origin', '*'); res.setHeader('Access-Control-Allow-Methods', 'GET, OPTIONS'); if (req.method === 'OPTIONS') { res.status(200).end(); return; } const { date } = req.query; const dateStr = date || new Date().toISOString().split('T')[0].replace(/-/g, ''); const SPORTS = [ { key: 'nba', sport: 'basketball', league: 'nba', label: 'NBA' }, { key: 'mlb', sport: 'baseball', league: 'mlb', label: 'MLB' }, { key: 'nhl', sport: 'hockey', league: 'nhl', label: 'NHL' }, ]; const allPicks = []; for (const s of SPORTS) { try { const url = `https://site.api.espn.com/apis/site/v2/sports/${s.sport}/${s.league}/scoreboard?dates=${dateStr}&limit=50`; const r = await fetch(url, { headers: { 'User-Agent': 'Mozilla/5.0', 'Accept': 'application/json' } }); if (!r.ok) continue; const data = await r.json(); const events = data.events || []; for (const e of events) { const comp = e.competitions?.[0]; if (!comp) continue; const home = comp.competitors?.find(c => c.homeAway === 'home'); const away = comp.competitors?.find(c => c.homeAway === 'away'); if (!home || !away) continue; const homeTeam = home.team?.displayName || ''; const awayTeam = away.team?.displayName || ''; const homeAbbr = home.team?.abbreviation || ''; const awayAbbr = away.team?.abbreviation || ''; // Get odds if available const odds = comp.odds?.[0]; const spreadVal = odds?.spread || null; const ouVal = odds?.overUnder || null; const homeML = odds?.homeTeamOdds?.moneyLine || null; const awayML = odds?.awayTeamOdds?.moneyLine || null; // Format spread const spread = spreadVal != null ? (spreadVal > 0 ? `+${spreadVal}` : String(spreadVal)) : null; const ou = ouVal ? String(ouVal) : null; const ml = homeML ? (homeML > 0 ? `+${homeML}` : String(homeML)) : null; const awayMLStr = awayML ? (awayML > 0 ? `+${awayML}` : String(awayML)) : null; // Game time const gameDate = new Date(e.date); const timeStr = gameDate.toLocaleTimeString('en-US', { hour: 'numeric', minute: '2-digit', timeZone: 'America/New_York' }) + ' ET'; // Network const network = comp.broadcasts?.[0]?.names?.[0] || ''; // Playoff/series info const playoff = e.season?.type === 3; const seriesInfo = comp.series ? `Series: ${comp.series.summary}` : null; // Injuries const injuries = []; for (const team of [home, away]) { const inj = team.team?.injuries || []; inj.slice(0, 2).forEach(i => { if (i.athlete?.displayName && i.status) { injuries.push(`${i.athlete.displayName} (${team.team?.abbreviation}) — ${i.status}: ${i.longComment || i.shortComment || 'listed'}`); } }); } // Run model const pick = generatePick({ id: `${s.key}_${e.id}`, sport: s.label, homeTeam, awayTeam, homeAbbr, awayAbbr, spread, ou, ml, awayML: awayMLStr, time: timeStr, network, playoff, seriesInfo, injuries, date: dateStr, }); if (pick.signal !== 'NEUTRAL') { allPicks.push(pick); } } } catch (err) { console.error(`Error fetching ${s.label}:`, err.message); } } return res.status(200).json({ date: dateStr, count: allPicks.length, picks: allPicks, }); } function generatePick(game) { const spread = parseFloat((game.spread || '0').replace('+', '')) || 0; const ml = parseFloat((game.ml || '0').replace('+', '')) || 0; const awayML = parseFloat((game.awayML || '0').replace('+', '')) || 0; // Implied win probability from ML const impliedWinPct = ml !== 0 ? (ml < 0 ? Math.abs(ml) / (Math.abs(ml) + 100) * 100 : 100 / (ml + 100) * 100) : 52; // Value score — detect line mispricing const spreadAbs = Math.abs(spread); const publicBias = spreadAbs > 10 ? 3 : spreadAbs > 6 ? 1.5 : 0; const seriesDisconnect = game.seriesInfo && impliedWinPct > 65 && spreadAbs < 5 ? 3 : 0; const injuryPenalty = (game.injuries || []).length * 0.5; const valueScore = (impliedWinPct - 52) * 0.08 + publicBias * 0.5 + seriesDisconnect - injuryPenalty; // Signal let signal, conf; if (valueScore >= 4) { signal = 'STRONG_BET'; conf = Math.min(84, 65 + valueScore * 1.1); } else if (valueScore >= 2) { signal = 'LEAN'; conf = Math.min(74, 56 + valueScore * 1.5); } else if (valueScore <= -2) { signal = 'FADE'; conf = Math.min(70, 54 + Math.abs(valueScore)); } else { signal = 'NEUTRAL'; conf = 52; } // Recommendation const favTeam = spread < 0 ? game.homeTeam.split(' ').pop() : game.awayTeam.split(' ').pop(); const dogTeam = spread < 0 ? game.awayTeam.split(' ').pop() : game.homeTeam.split(' ').pop(); let rec; if (signal === 'STRONG_BET' || signal === 'LEAN') { rec = spread !== 0 ? `${favTeam} ${game.spread}` : `${favTeam} ML`; } else if (signal === 'FADE') { rec = `Fade ${favTeam} — take ${dogTeam} ${spread !== 0 ? (spread < 0 ? `+${Math.abs(spread)}` : spread) : 'ML'}`; } else { rec = `Pass — ${game.awayTeam.split(' ').pop()} @ ${game.homeTeam.split(' ').pop()}`; } // Analysis const analyses = { STRONG_BET: `Strong model edge detected on ${favTeam}. Implied win probability of ${Math.round(impliedWinPct)}% combined with line movement signals clear value. Public bias appears to be creating an exploitable gap on the other side.`, LEAN: `Moderate edge on ${favTeam}. Line appears slightly mispriced relative to actual team quality. Worth monitoring for further movement before game time.`, FADE: `Model suggests fading ${favTeam} at this number. Line overvalues public perception. ${dogTeam} at ${spreadAbs > 0 ? `+${spreadAbs}` : 'even'} represents value.`, NEUTRAL: `No significant edge detected. Line appears fairly priced. Model suggests passing.`, }; // Factors const factors = []; if (impliedWinPct > 65) factors.push({ t: `${Math.round(impliedWinPct)}% implied win prob`, p: signal !== 'FADE' }); if (spreadAbs > 7) factors.push({ t: `Large spread — public darling risk`, p: signal === 'FADE' }); if (game.seriesInfo) factors.push({ t: game.seriesInfo, p: signal === 'STRONG_BET' || signal === 'LEAN' }); if (game.playoff) factors.push({ t: 'Playoff intensity factor', p: true }); if (game.injuries?.length) factors.push({ t: `${game.injuries.length} injury flag${game.injuries.length > 1 ? 's' : ''}`, p: false }); if (signal === 'STRONG_BET') factors.push({ t: 'Sharp money thesis aligned', p: true }); if (factors.length < 2) factors.push({ t: 'Line movement signal', p: signal !== 'FADE' }); return { id: game.id, date: new Date().toLocaleDateString('en-US', { month: 'numeric', day: 'numeric', year: '2-digit' }), sport: game.sport, away: game.awayTeam, home: game.homeTeam, time: game.time, network: game.network, spread: game.spread || '—', ou: game.ou || '—', ml: game.ml || '—', awayML: game.awayML || '—', playoff: game.playoff, seriesInfo: game.seriesInfo, injuries: game.injuries || [], signal, rec, conf: Math.round(conf), predSpread: +(spread + valueScore * 0.3).toFixed(1), analysis: analyses[signal], factors, result: '', units: '1', }; }
